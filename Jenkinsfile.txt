pipeline {
    agent any

    stages {
        stage('Build') {
            steps {
                bat "${mvnHome}/bin/mvn clean compile"
            }
        }
        stage('Test') {
            steps {
                bat 'mvn test'
            }
        }
        stage('Generate Allure Report') {
            steps {
                bat 'mvn allure:report'
            }
        }
    }
    post {
        always {
            allure([
                includeProperties: false,
                jdk: '',
                results: [[path: 'target/allure-results']]
            ])
        }
    }
}
